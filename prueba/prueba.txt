<!DOCTYPE html>
<html lang="es">
<head>
  <meta charset="UTF-8">
  <title>Productos</title>
</head>
<body>
  <h2>Productos con precio mayor a $100</h2>
  <div id="productos"></div>

  <h2>Nombres de todos los productos en minúscula</h2>
  <div id="nombres-minuscula"></div>

  <script>
    // Arreglo de productos
    const productos = [
      { nombre: 'Teclado', precio: 120 },
      { nombre: 'Mouse', precio: 80 },
      { nombre: 'Monitor', precio: 300 },
      { nombre: 'Parlantes', precio: 90 },
      { nombre: 'Laptop', precio: 1500 }
    ];

    // 🔹 Productos con precio mayor a $100
    const productosCaros = productos.filter(producto => producto.precio > 100);

    // 🔹 Nombres en minúscula
    const nombresMinuscula = productos.map(producto => producto.nombre.toLowerCase());

    // 🔹 Mostrar productos caros en el DOM
    const contenedorProductos = document.getElementById('productos');

    productosCaros.forEach(producto => {
      const item = document.createElement('p');
      item.textContent = `${producto.nombre}: $${producto.precio}`;
      contenedorProductos.appendChild(item);
    });

    // 🔹 Mostrar nombres en minúscula en el DOM
    const contenedorNombres = document.getElementById('nombres-minuscula');

    nombresMinuscula.forEach(nombre => {
      const item = document.createElement('p');
      item.textContent = nombre;
      contenedorNombres.appendChild(item);
    });
  </script>
</body>
</html>
